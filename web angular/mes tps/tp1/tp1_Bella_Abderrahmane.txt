function tellBob(phrase: string) {
 if (phrase.endsWith('?')) {
 return "bien sur";
 } else if (phrase.trim() === '') {
 return "ok, si tu veux";
 } else if(phrase.toLocaleUpperCase() === phrase) {
 return "du calme"

 return "peu m'importe"
 }
}

 Pangramme
const alphabet: string = "abcdefghijklmnopqrstuvwxyz"
function isPangram(phrase: string) : boolean {
 for(let lettre of alphabet) {
 if (phrase.indexOf(lettre) === -1) {
 return false;
 }
 }
 return true;
}
console.log(isPangram("eez"));
console.log(isPangram(alphabet));
console.log(isPangram("portez ce vieux whisky au juge blond qui
fume"));

Les allergies
console.clear()
const allergies = {
 oeuf: 1,
 cacahuete: 2,
 "fruit de mer":4,
 fraise: 8,
 tomate: 16,
 chocolat: 32,
 pollen: 64,
 chats: 128
}
console.log(Object.entries(allergies).reverse());
class Patient {
 private patientAllergies: string [] = []
 constructor(patientScore: number) {
 for (let [allergie, score] of Object.entries(allergies).reverse()) {
 if (score <= patientScore) {
 this.patientAllergies.push(allergie);
 patientScore -= score
 }
 }
 }
 isAllergicTo(item: string) : boolean {
 return this.patientAllergies.includes(item);
 }
 listAllergies(): string[] {
 return this.patientAllergies;
 }
}
const patient: Patient = new Patient(34);
console.log(patient.listAllergies())
console.log(patient.isAllergicTo("chat"))
console.log(patient.isAllergicTo("cacahuete"))
